template<class Key, AZStd::size_t FixedNumBuckets, AZStd::size_t FixedNumElements, AZStd::size_t NumLocks = 8, class Hasher = AZStd::hash<Key>, class EqualKey = AZStd::equal_to<Key>>
class AZStd::concurrent_fixed_unordered_set< Key, FixedNumBuckets, FixedNumElements, NumLocks, Hasher, EqualKey >

Concurrent fixed unordered set container, similar to fixed_unordered_set, but uses a striped array of locks to allow concurrent access.
